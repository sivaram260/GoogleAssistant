#include < ESP8266WiFi.h>
#include "Adafruit_MQTT.h"
#include "Adafruit_MQTT_Client.h"
#define wifi"your wifi name"
#define password "Your wifi password"
#define server "io.adafruit.com"
#define port 1883
#define username "your adafruit username"
#define key "your adafruit key"
WiFiClient esp;
Adafruit_MQTT_Client mqtt(&esp,server,port,username,key);
Adafruit_MQTT_Subscribe feed(&mqtt,username"/feeds/feed name");
void setup() {

Serial.begin(115200);
pinMode(D2,OUTPUT);
delay(10);
Serial.println("Adafruit MQTT");
Serial.println();
Serial.println();
Serial.println("connecting to ");
Serial.println(wifi);
WiFi.begin(wifi,password);
while (WiFi.status()!=WL_CONNECTED)
{
delay(500);
Serial.print(".");
}
Serial.println();
Serial.println("WiFi connected");
Serial.println("IP address: ");
mqtt.subscribe(&feed);
Serial.println("Connecting to MQTT");
while(mqtt.connect())
{
Serial.print(".");
}
}
void loop() {
Serial.println(" ");
while(mqtt.readSubscription(10000))
{
Serial.print("Got: ");
Serial.print((char*)feed.lastread);
int a =atoi((char*)feed.lastread);
Serial.print('/t');
Serial.println(a);
if(a==1)
{
digitalWrite(D2,0);
}
else
{
digitalWrite(D2,1);
}
}
}
